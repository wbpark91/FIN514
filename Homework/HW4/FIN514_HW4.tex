\begin{homeworkProblem}
\begin{enumerate}[a.]
	\item		%% a.
	By Ito's lemma,
	\begin{equation*}
	\begin{aligned}
		df(X) 	&= \itox{f}{X}	\\
				&= 2X(t)dX(t) + \frac{1}{2} \times 2dt	\\
		\Rightarrow X(t)dX(t) &= \frac{1}{2}df(X) - \frac{1}{2}dt	\\
		\Rightarrow \int_0^t X(\tau)d\tau &= \frac{1}{2}\int_0^t dX^2(\tau) + \int_0^t \frac{1}{2}d\tau	\\
								&= \frac{1}{2}X^2(t) + \frac{1}{2}t
	\end{aligned}
	\end{equation*}	
	\item		%% b.
	Let $g(t, X) = tX(t)$. Then by Ito's lemma,
	\begin{equation*}
	\begin{aligned}
		dg(t, X) 	&= \itotx{g}{X}	\\
				&= X(t)dt + tdX(t)	\\
		\Rightarrow tdX(t) &= d(tX(t)) - X(t)dt	\\
		\Rightarrow \int_0^t \tau dX(\tau) &= tX(t) - \int_0^t X(\tau)d\tau
	\end{aligned}
	\end{equation*}
	\item		%% c.
	Let $h(X) = X^3(t)$. Then by Ito's lemma,
	\begin{equation*}
	\begin{aligned}
		dh(X) 	&= \itox{h}	{X}	\\
				&= 3X^2(t)dX(t) + \frac{1}{2}6X(t)dt	\\
				&= 3X^2(t)dX(t) + 3X(t)dt	\\
		\Rightarrow X^2(t)dX(t) &= \frac{1}{3}d(X^3(t)) - X(t)dt	\\
		\Rightarrow \int_0^t X^2(\tau)dX(\tau) &= \frac{1}{3}X^3(t) - \int_0^t X(\tau)d\tau
	\end{aligned}
	\end{equation*}
\end{enumerate}
\end{homeworkProblem}

\begin{homeworkProblem}
	By Ito's lemma,
	\begin{equation*}
	\begin{aligned}
		dF 	&= \itotx{F}{S}	\\
			&= -(r - \delta)Fdt + \frac{F}{S}dS(t)	\\
	\end{aligned}
	\end{equation*}
	Since $dS(t) = (\mu - \delta)S(t)dt + \sigma S(t)dX(t)$,
	\begin{equation*}
	\begin{aligned}
		dF 	&= -(r - \delta)Fdt + \frac{F}{S}((\mu - \delta)S(t)dt + \sigma S(t)dX(t))	\\
			&= -(r - \delta)Fdt + (\mu - \delta)Fdt + \sigma FdX(t)	\\
			&= (\mu - r)Fdt + \sigma FdX(t)
	\end{aligned}
	\end{equation*}
\end{homeworkProblem}

\begin{homeworkProblem}
\begin{enumerate}[a.]
	\item		%% a.
	By Ito's product rule,
	\begin{equation*}
	\begin{aligned}
		dY 	&= d\left(\frac{S_2}{S_1}\right)	\\
			&= \frac{1}{S_1}dS_2 + S_2 d\left(\frac{1}{S_1}\right) + dS_2 d\left(\frac{1}{S_1}\right)	
	\end{aligned}
	\end{equation*}
	By Ito's lemma,
	\begin{equation*}
	\begin{aligned}
		d\left(\frac{1}{S_1}\right) 	&= -\frac{1}{S^2_1}dS_1 + \frac{1}{2} \times 2(dS_1)^2	\\
							&= -\frac{1}{S^2_1}(\mu_1 S_1 dt + \sigma_1 S_1 dX_1) + \frac{1}{S^3_1}\sigma^2_1 S^2_1 dt	\\
							&= -\frac{\mu_1}{S_1}dt - \frac{\sigma_1}{S_1}dX_1 + \frac{\sigma^2_1}{S_1}dt	\\
							&= \frac{-\mu_1 + \sigma^2_1}{S_1}dt - \frac{\sigma_1}{S_1}dX_1
	\end{aligned}
	\end{equation*}
	Therefore,
	\begin{equation*}
	\begin{aligned}
		dY 	&= \frac{1}{S_1}(\mu_2 S_2 dt + \sigma_2 S_2 dX_2) + S_2\left(\frac{-\mu_1 + \sigma^2_1}{S_1}dt - \frac{\sigma_1}{S_1}dX_1\right) + (\mu_2 S_2 dt + \sigma_2 S_2 dX_2) \left(\frac{-\mu_1 + \sigma^2_1}{S_1}dt - \frac{\sigma_1}{S_1}dX_1\right)	\\
			&= \mu_2 Y dt + \sigma_2 YdX_2 + (-\mu_1 + \sigma^2_1)Ydt - \sigma_1 Y dX_1 - \sigma_1 \sigma_2 \rho Ydt ~~ (\because dX_1 dX_2 = \rho dt)
	\end{aligned}
	\end{equation*}
	Let $\sigma_3 X_3$ = $\sigma_1 X_1 + \sigma_2 X_2$, where $\sigma_3 = \sqrt{\sigma_1^2 + \sigma_2^2 + 2\rho \sigma_1 \sigma_2}$. Since $X_1$ and $X_2$ have expected value zero, $\E[X_3]$ also equal to zero. Furthermore, $\Var[X_3] = \Var[\sigma_1 X_1 + \sigma_2 X_2] / \sigma_3^2 = (\sigma_1^2 + \sigma_2^2 + 2\rho \sigma_1 \sigma_2)t / \sigma_3^2 = t$. Therefore, $\sigma_3 X_3 = \sigma_1 X_1 + \sigma_2 X_2$. Finally $dY$ can be represented as follows.
	\begin{equation*}
	\begin{aligned}
		dY 	&= (\mu_2 - \mu_1 + \sigma_1^2 - \sigma_1 \sigma_2 \rho)Ydt + \sigma_3 Y dX_3	\\
			&= \mu_Y Y_t dt + \sigma_Y Y_t dX_{3t}	\\
		\text{where~} \mu_Y &= \mu_2 - \mu_1 + \sigma_1^2 - \sigma_1 \sigma_2 \rho, \sigma_Y = \sqrt{\sigma_1^2 + \sigma_2^2 + 2\rho \sigma_1 \sigma_2}
	\end{aligned}
	\end{equation*}
	\item		%% b.
	By Ito's product rule,
	\begin{equation*}
	\begin{aligned}
		dZ_t 		&= d(Y_t B_{Et})	\\
				&= B_{Et}dY_t + Y_tdB_{Et} + dY_t dB_{Et}	\\
				&= B_E{t}(\mu_Y Y_t dt + \sigma_Y Y_t dX_{3t}) + Y_t(r_EB_{Et}dt),	~~ dB_{Et} = r_EB_{Et}dt ~~ \text{because the bond is riskless.}	\\
				&= (\mu_Y + r_E)Z_t dt + \sigma_Y Z_t dX_{3t}	\\
				&= (\mu_Z)Z_t dt + \sigma_Y Z_t dX_{3t}	\\
				&\text{where} ~~ \mu_Z = \mu_Y + r_E.
	\end{aligned}
	\end{equation*}
\end{enumerate}
\end{homeworkProblem}
